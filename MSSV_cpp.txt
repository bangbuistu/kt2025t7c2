#include<iostream>
using namespace std;
bool kiemTraSNT(int n)
{
	if(n<2)
		return false;
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
			return false;
	}
	return true;
}

int main()
{
	int a=10;
	if(kiemTraSNT(a)==true)
		cout<<"La so nguyen to";
	else
		cout<<"Khong phai so nguyen to";
	system("pause");
	return 0;
}
